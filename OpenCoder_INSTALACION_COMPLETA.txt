                    2. INSTALACIÓN RÁPIDA (RECOMENDADA)
================================================================================

Opción A: Instalación automática (Linux/macOS)

    # Clonar repositorio
    git clone https://github.com/Asphyksia/OpenCoder.git
    cd OpenCoder
    
    # Ejecutar instalador
    chmod +x install.sh
    ./install.sh
    
    # Configurar API key (el script te preguntará)
    # O editar manualmente: nano .env
    
    # Iniciar
    ./start.sh

Opción B: Instalación con uv (ultra rápido)

    # Clonar
    git clone https://github.com/Asphyksia/OpenCoder.git
    cd OpenCoder
    
    # Instalar uv si no lo tienes
    curl -LsSf https://astral.sh/uv/install.sh | sh
    
    # Crear venv e instalar todo
    uv venv --python 3.12
    source .venv/bin/activate  # Linux/macOS
    # .venv\Scripts\activate   # Windows
    
    uv pip install -r requirements.txt
    uv pip install 'aider-chat[all]'
    
    # Frontend
    cd frontend && npm install && cd ..
    
    # Configurar
    cp .env.example .env
    # Editar .env con tu API key
    
    # Iniciar
    ./start.sh

                    4. INSTALACIÓN MANUAL PASO A PASO
================================================================================

PASO 1: Verificar Python correcto

    python3 --version
    # Debe mostrar 3.10.x, 3.11.x o 3.12.x
    
    # Si tienes Python 3.13:
    # Linux: sudo apt install python3.12 python3.12-venv

PASO 2: Clonar y crear entorno virtual

    git clone https://github.com/Asphyksia/OpenCoder.git
    cd OpenCoder
    
    # Linux/macOS
    python3.12 -m venv venv
    source venv/bin/activate
    
    # Windows (PowerShell)
    py -3.12 -m venv venv
    .\venv\Scripts\Activate.ps1
    
    # Windows (Git Bash)
    python -m venv venv
    source venv/Scripts/activate

PASO 3: Instalar dependencias Python

    pip install --upgrade pip
    pip install -r requirements.txt
    pip install 'aider-chat[all]'

PASO 4: Instalar dependencias Frontend

    cd frontend
    npm install
    cd ..

PASO 5: Configurar variables de entorno

    cp .env.example .env
    
    # Editar .env con tu API key
    nano .env  # o code .env

PASO 6: Iniciar servicios

    # Opción A: Script unificado
    ./start.sh
    
    # Opción B: Manual
    # Terminal 1 - Backend
    source venv/bin/activate
    uvicorn opencoder.api.main:app --reload --port 8000
    
    # Terminal 2 - Frontend
    cd frontend && npm run dev

NORMALIZACIÓN DE MODELOS:

OpenCoder normaliza automáticamente los nombres de modelos:

    openai/Qwen/Qwen3-Coder     → Qwen/Qwen3-Coder
    anthropic/claude-3-opus     → claude-3-opus
    ollama/codellama            → codellama
    
    # Pero mantiene vendor prefixes:
    Qwen/Qwen3-Coder            → Qwen/Qwen3-Coder (sin cambios)
    deepseek-ai/DeepSeek-V3.1   → deepseek-ai/DeepSeek-V3.1 (sin cambios)

PROBLEMA: Python 3.13 detectado

    SOLUCIÓN: Instalar Python 3.12
    
    # Linux
    sudo apt install python3.12 python3.12-venv
    python3.12 -m venv venv
    
    # Mac
    brew install python@3.12
    python3.12 -m venv venv

PROBLEMA: Error "litellm blocked by OpenGPU"

    SOLUCIÓN: El patch ya está incluido en opencoder/core/litellm_patch.py
    
    # Verificar que se aplica automáticamente
    python -c "from opencoder.core.litellm_patch import apply_patch; apply_patch()"

PROBLEMA: Aider no puede conectar con OpenGPU

    SOLUCIÓN: Verificar variables de entorno
    
    # En .env o exportar:
    export OPENAI_API_KEY=$OPENGPU_API_KEY
    export OPENAI_API_BASE=https://relaygpu.com/backend/openai/v1

PROBLEMA: Puerto 8000 o 3000 ocupado

    SOLUCIÓN: Usar puertos diferentes
    
    # Backend en 8001
    uvicorn opencoder.api.main:app --port 8001
    
    # Frontend en 3001
    PORT=3001 npm run dev

PROBLEMA: Error "No module named 'opencoder'"

    SOLUCIÓN: Ejecutar desde el directorio correcto o instalar
    
    # Opción A: Desde el directorio del proyecto
    cd OpenCoder
    python -m uvicorn opencoder.api.main:app
    
    # Opción B: Instalar en modo desarrollo
    pip install -e .

PROBLEMA: Docker no puede acceder a archivos locales

    SOLUCIÓN: Montar volumen correctamente
    
    docker run -v /ruta/a/tu/proyecto:/app/workspace ...

NORMALIZACIÓN DE MODELOS (opencoder/core/litellm_patch.py):

    def normalize_model_name(model: str) -> str:
        """
        Normaliza nombres de modelos para OpenGPU Relay.
        
        Reglas:
        1. Quitar prefijos conocidos: openai/, anthropic/, ollama/
        2. Mantener vendor prefixes: Qwen/, deepseek-ai/
        3. El resultado debe coincidir con los nombres de OpenGPU
        """
        prefixes = ['openai/', 'anthropic/', 'ollama/']
        for prefix in prefixes:
            if model.lower().startswith(prefix):
                return model[len(prefix):]
        return model

COMPATIBILIDAD MULTIPLATAFORMA (scripts/aider_wrapper.py):

    # Detectar Python 3.12 en cualquier sistema
    import platform
    import shutil
    
    def find_python():
        system = platform.system()
        if system == "Linux":
            candidates = ['python3.12', 'python3.11', 'python3']
        elif system == "Darwin":  # macOS
            candidates = ['python3.12', 'python3.11', 'python3']
        elif system == "Windows":
            candidates = ['py', 'python', 'python3']
        
        for cmd in candidates:
            if shutil.which(cmd):
                return cmd
        return None

LITELLM PATCH (opencoder/core/litellm_patch.py):

    # Aplicar al inicio de la aplicación
    from opencoder.core.litellm_patch import apply_patch
    apply_patch()

AIDER BRIDGE (opencoder/core/aider_bridge.py):

    # Usar normalización consistente
    from opencoder.core.litellm_patch import normalize_model_name
    
    model = normalize_model_name(model)
